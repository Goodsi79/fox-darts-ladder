    if (!enableLeg3) {

      if (el.type === "checkbox") el.checked = false;

  [1,2].forEach(n => {

    const w = document.querySelector(`.leg-winner[data-leg="${n}"]`);

    const c45 = document.querySelector(`.leg-45[data-leg="${n}"]`);

    if (w) w.disabled = false;

    if (c45) c45.disabled = false;

    if (b) b.disabled = false;

  });

}

/***** SUBMIT MATCH – with correct history mapping + nicknames *****/

function submitMatch() {

  const msg = document.getElementById("msg");

  msg.textContent = "";

  msg.className = "message message-info";


  const c = parseInt(document.getElementById("challenger").value);

  const o = parseInt(document.getElementById("opponent").value);

  if (isNaN(c) || isNaN(o)) {

    msg.textContent = "Select challenger and opponent.";

    msg.className = "message message-error";

    return;

  }


  // Capture names and nicknames BEFORE any ladder reordering

  const challengerName = ladder[c].name;

  const challengerNick = ladder[c].nick || null;

  const opponentName = ladder[o].name;

  const opponentNick = ladder[o].nick || null;


  const legs = getLegsFromDOM();

  const cLegs = legs.filter(l => l.winner === "challenger").length;

  const oLegs = legs.filter(l => l.winner === "opponent").length;

  const totalPlayed = legs.filter(l => l.winner).length;


  // Validation: first to 2, max 3 legs

  if (!(cLegs === 2 || oLegs === 2)) {

    msg.textContent = "Match must end 2–0 or 2–1. One player must have exactly 2 legs.";

    msg.className = "message message-error";

    return;

  }


  if (totalPlayed < 2 || totalPlayed > 3) {

    msg.textContent = "Invalid leg structure. Must be 2–0 (2 legs) or 2–1 (3 legs).";

    msg.className = "message message-error";

    return;

  }


  if (totalPlayed === 3 && !((cLegs === 2 && oLegs === 1) || (cLegs === 1 && oLegs === 2))) {

    msg.textContent = "Invalid 2–1 structure.";

    msg.className = "message message-error";

    return;

  }


  if (totalPlayed === 2 && !((cLegs === 2 && oLegs === 0) || (cLegs === 0 && oLegs === 2))) {

    msg.textContent = "Invalid 2–0 structure.";

    msg.className = "message message-error";

    return;

  }


  // 45-dart rule

  for (let i = 0; i < legs.length; i++) {

    const l = legs[i];

    if (l.reached45 && !l.bull) {

      msg.textContent = `Leg ${i+1} reached 45 darts but nearest bull not confirmed.`;

      msg.className = "message message-error";

      return;

    }

  }


  // Winner / loser

  const winnerIndex = cLegs > oLegs ? c : o;

  const loserIndex  = cLegs > oLegs ? o : c;

  const winnerName = ladder[winnerIndex].name;

  const loserName  = ladder[loserIndex].name;


  const winnerLegs = Math.max(cLegs, oLegs);

  const loserLegs  = Math.min(cLegs, oLegs);


  // Stats

  ladder[winnerIndex].w++;

  ladder[loserIndex].l++;

  ladder[winnerIndex].lf += winnerLegs;

  ladder[winnerIndex].la += loserLegs;

  ladder[loserIndex].lf  += loserLegs;

  ladder[loserIndex].la  += winnerLegs;


  const stamp = new Date().toLocaleString();

  ladder[winnerIndex].last = `W vs ${loserName} (${stamp})`;

  ladder[loserIndex].last  = `L vs ${winnerName} (${stamp})`;


  // Movement

  let movementText = "No movement";

  if (winnerIndex > loserIndex) {

    const oldPos = winnerIndex + 1;

    const newPos = loserIndex + 1;

    const moved = ladder.splice(winnerIndex,1)[0];

    ladder.splice(loserIndex,0,moved);

    movementText = `Moved from ${oldPos} to ${newPos}`;

  }


  // Match record with correct challenger/opponent mapping + nicknames

  const matchRecord = {

    challenger: challengerName,

    challengerNick: challengerNick,

    opponent: opponentName,

    opponentNick: opponentNick,

    winner: winnerName,

    loser: loserName,

    legs: legs,

    timestamp: stamp,

    movement: movementText

  };


  db.ref("history").push(matchRecord);

  saveLadder();


  msg.textContent = "Match recorded and ladder updated.";

  msg.className = "message message-info";


  // Reset leg UI

  document.querySelectorAll(".leg-winner").forEach(sel => {

    sel.value = "";

    sel.disabled = false;

  });

  document.querySelectorAll(".leg-45").forEach(cb => {

    cb.checked = false;

    cb.disabled = false;

  });

  document.querySelectorAll(".leg-bull").forEach(cb => {

    cb.checked = false;

    cb.disabled = false;

  });

  [1,2,3].forEach(n => {

    const bullArea = document.getElementById(`bull-${n}`);

    if (bullArea) bullArea.style.display = "none";

  });


  // Leg 3 should start disabled again

  document.querySelector('.leg-winner[data-leg="3"]').disabled = true;

  document.querySelector('.leg-45[data-leg="3"]').disabled = true;

  document.querySelector('.leg-bull[data-leg="3"]').disabled = true;


  recalcLegState();

}


/* Export ladder + history */

function exportAll() {

  const data = {

    ladder: ladder,

    history: historyCache

  };

  const dataStr = JSON.stringify(data, null, 2);

  const blob = new Blob([dataStr], { type: "application/json" });

  const url = URL.createObjectURL(blob);


  const a = document.createElement("a");

  a.href = url;

  a.download = "fox-darts-ladder-and-history-backup.json";

  a.click();


  URL.revokeObjectURL(url);

  document.getElementById("admin-msg").textContent = "Exported ladder + history.";

}


/* Import ladder + history */

function importAll(event) {

  const file = event.target.files[0];

  const msg = document.getElementById("admin-msg");


  if (!file) {

    msg.textContent = "No file selected.";

    return;

  }


  const reader = new FileReader();

  reader.onload = function(e) {

    try {

      const imported = JSON.parse(e.target.result);


      if (typeof imported !== "object" || imported === null) {

        msg.textContent = "Invalid file format. Expected an object with ladder and history.";

        return;

      }


      if (!Array.isArray(imported.ladder) || typeof imported.history !== "object") {

        msg.textContent = "Invalid file format. Missing ladder or history.";

        return;

      }


      if (!confirm("Importing will replace the current ladder and history for everyone. Continue?")) {

        msg.textContent = "Import cancelled.";

        return;

      }


      ladder = imported.ladder.map(p => ({

        ...p,

        nick: p.nick || null,

        delta: 0

      }));

      previousLadder = JSON.parse(JSON.stringify(ladder));

      historyCache = imported.history;


      db.ref("ladder").set(ladder);

      db.ref("history").set(historyCache);


      renderLadder();

      renderSelects();


      msg.textContent = "Ladder and history imported successfully.";

    } catch (err) {

      msg.textContent = "Error reading file. Ensure it is a valid JSON backup.";

    }

  };


  reader.readAsText(file);

}
